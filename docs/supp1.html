Boxplot

degenes<-lapply(res, function(x) {
  tryCatch({
    y<-x[[2]]
    y<-na.omit(y)
    data.frame(y)%>%filter(padj<0.1)%>%nrow()}, 
    error=function(err) {NA})
})

boxplot<-lapply(unique(Idents(fgf.neur.unmap)), function(x) {
  z<-unlist(degenes[grep(paste0("^",x,"_"), names(degenes))])
})
